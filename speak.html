This page will listen to what you say and attempt to repeat it back to you.

You must allow access to your microphone. Open your console for debug info. This will not work on a mobile browser.

  • https://github.com/TalAter/annyang
  • https://github.com/kripken/speak.js

if (annyang) {
  	
  var sayThis = function(repeat) {
  	speak(repeat);
  }

  // Let's define a command.
  var commands = {
    '*repeat': sayThis
  };

  // Debug info for the console
  annyang.debug();

  // Initialize annyang with our commands
  annyang.init(commands);

  // Start listening.
  annyang.start();
}
